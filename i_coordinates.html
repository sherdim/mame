def lepestok(l, w, color='#795BD3', alpha=0.8, x0=0.0, y0=0.0):
    '''—Ä–∏—Å—É–µ–º –ª–µ–ø–µ—Å—Ç–æ–∫
    
    –ø–∞—Ä–∞–º–µ—Ç—Ä—ã:
    x0, y0 - –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã —Ü–µ–Ω—Ç—Ä–∞
    color - —Ü–≤–µ—Ç –∑–∞–ª–∏–≤–∫–∏
    alpha - —Å—Ç–µ–ø–µ–Ω—å –ø—Ä–æ–∑—Ä–∞—á–Ω–æ—Å—Ç–∏
    '''
    xx=[x0, x0-w/2, x0, x0+w/2]
    yy=[y0-l/2, y0, y0+l/2, y0]
    h = fill(xx,yy, color, alpha=alpha); 
    
    axis('equal'); #–æ–¥–∏–Ω–∞–∫–æ–≤—ã–π —à–∞–≥ —Å–µ—Ç–∫–∏ –ø–æ –æ—Å—è–º
    return h
